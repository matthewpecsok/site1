💻 SQL Query Examples

1. Basic SELECT Statement

SELECT first_name, last_name, email FROM students WHERE enrollment_date >= '2026-01-01' ORDER BY last_name;

2. JOIN Multiple Tables

SELECT s.first_name, s.last_name, c.course_name, e.grade FROM students s JOIN enrollments e ON s.student_id = e.student_id JOIN courses c ON e.course_id = c.course_id WHERE c.department = 'Information Systems';

3. Aggregate Functions

SELECT c.department, COUNT(*) as total_courses, AVG(c.credits) as avg_credits FROM courses c GROUP BY c.department HAVING COUNT(*) > 3;

🌐 Node.js Database Connection Example

const mysql = require('mysql2'); // Create connection pool const pool = mysql.createPool({ host: process.env.DB_HOST, user: process.env.DB_USER, password: process.env.DB_PASSWORD, database: process.env.DB_NAME, waitForConnections: true, connectionLimit: 10, queueLimit: 0 }); // API endpoint to get all students app.get('/api/students', async (req, res) => { try { const [rows] = await pool.execute( 'SELECT student_id, first_name, last_name, email FROM students' ); res.json(rows); } catch (error) { console.error('Database error:', error); res.status(500).json({ error: 'Internal server error' }); } });

📱 Frontend AJAX Call

// Fetch students from API async function loadStudents() { try { const response = await fetch('/api/students'); const students = await response.json(); const studentList = document.getElementById('student-list'); studentList.innerHTML = students.map(student => ` <div class="student-card"> <h3>${student.first_name} ${student.last_name}</h3> <p>Email: ${student.email}</p> </div> `).join(''); } catch (error) { console.error('Error loading students:', error); } }

🔐 Security Best Practices

  • Use parameterized queries to prevent SQL injection
  • Never store passwords in plain text - use proper hashing
  • Implement input validation on both client and server
  • Use environment variables for database credentials
  • Regular database backups are essential for data protection
  • Limit database user privileges to only what's necessary
